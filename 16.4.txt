SELECT I.SUMMAR, I.DESCRIPTION, il.name FROM issues I JOIN issueslists IL ON I.ISSUESLIST_ID = IL.ID;
SELECT I.SUMMAR, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME FROM issues I JOIN users U on I.USER_ID_ASSIGNEDTO = U.ID;
SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) as iloscZadan FROM users U join issues I on U.ID = I.USER_ID_ASSIGNEDTO GROUP BY U.ID HAVING COUNT(*) > 0;
